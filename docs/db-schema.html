    %% ──────────── IDENTITY ────────────
    auth_users {
        uuid id PK
    }

    profiles {
        uuid id PK "= auth.users.id"
        text username UK "unique, lowercase"
        text display_name
        text avatar_url
        timestamptz created_at
        timestamptz updated_at
    }

    user_preferences {
        uuid id PK
        uuid user_id FK "→ profiles.id"
        int focus_duration_minutes "default 25"
        int short_break_minutes "default 5"
        int long_break_minutes "default 15"
        int long_break_interval "default 4"
        text timezone
        bool reflection_enabled "default true"
        timestamptz created_at
        timestamptz updated_at
    }

    %% ──────────── GOALS ────────────
    long_term_goals {
        uuid id PK
        uuid user_id FK "→ profiles.id"
        text title
        text description
        goal_status status "active|completed|retired"
        timestamptz created_at
        timestamptz updated_at
    }

    process_goals {
        uuid id PK
        uuid user_id FK "→ profiles.id"
        uuid long_term_goal_id FK "→ long_term_goals.id"
        text title
        recurrence_type recurrence "daily|weekly"
        int target_sessions
        goal_status status "active|completed|retired"
        timestamptz created_at
        timestamptz updated_at
    }

    %% ──────────── SESSIONS ────────────
    sessions {
        uuid id PK
        uuid user_id FK "→ profiles.id"
        uuid process_goal_id FK "→ process_goals.id (nullable)"
        int planned_duration_minutes
        int actual_duration_seconds
        bool completed
        abandonment_reason abandonment_reason "nullable"
        focus_quality focus_quality "nullable"
        distraction_type distraction_type "nullable"
        text reflection_note "nullable"
        timestamptz started_at
        timestamptz ended_at
        timestamptz created_at
    }

    breaks {
        uuid id PK
        uuid user_id FK "→ profiles.id"
        uuid session_id FK "→ sessions.id"
        break_type type "short|long"
        int planned_duration_minutes
        int actual_duration_seconds
        break_usefulness usefulness "nullable: yes|somewhat|no"
        timestamptz started_at
        timestamptz ended_at
        timestamptz created_at
    }

    %% ──────────── SOCIAL ────────────
    friend_requests {
        uuid id PK
        uuid sender_id FK "→ profiles.id"
        uuid recipient_id FK "→ profiles.id"
        request_status status "pending|accepted|declined"
        timestamptz created_at
        timestamptz updated_at
    }

    friendships {
        uuid id PK
        uuid user_id FK "→ profiles.id"
        uuid friend_id FK "→ profiles.id"
        timestamptz created_at
    }

    encouragement_taps {
        uuid id PK
        uuid sender_id FK "→ profiles.id"
        uuid recipient_id FK "→ profiles.id"
        timestamptz created_at
    }

    %% ──────────── DEVICES ────────────
    devices {
        uuid id PK
        uuid user_id FK "→ profiles.id"
        text hardware_id UK "unique BLE device"
        text device_name
        text expo_push_token "nullable"
        timestamptz last_synced_at
        timestamptz created_at
        timestamptz updated_at
    }

    device_sync_log {
        uuid id PK
        uuid device_id FK "→ devices.id"
        sync_direction direction "up|down"
        text entity_type
        uuid entity_id
        timestamptz synced_at
    }

    %% ──────────── RELATIONSHIPS ────────────
    auth_users ||--|| profiles : "1:1 identity"
    profiles ||--|| user_preferences : "1:1 settings"
    profiles ||--o{ long_term_goals : "has many"
    profiles ||--o{ process_goals : "has many"
    long_term_goals ||--o{ process_goals : "has many"
    profiles ||--o{ sessions : "records"
    process_goals ||--o{ sessions : "tracks"
    sessions ||--o{ breaks : "followed by"
    profiles ||--o{ friend_requests : "sends"
    profiles ||--o{ friendships : "has"
    profiles ||--o{ encouragement_taps : "sends/receives"
    profiles ||--o{ devices : "owns"
    devices ||--o{ device_sync_log : "logs"
